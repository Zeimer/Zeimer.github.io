Zdefiniuj liczby konaturalne oraz ich relację bipodobieństwa. Pokaż, że jest to relacja równoważności.


Lemma sim_refl :
  forall n : conat, sim n n.

Lemma sim_sym :
  forall n m : conat, sim n m -> sim m n.

Lemma sim_trans :
  forall a b c : conat, sim a b -> sim b c -> sim a c.

Dzięki poniższemu będziemy mogli używac taktyki rewrite do przepisywania sim tak samo jak =.

Require Import Setoid.

Instance Equivalence_sim : Equivalence sim.
Proof.
  esplit; red.
    apply sim_refl.
    apply sim_sym.
    apply sim_trans.
Defined.
